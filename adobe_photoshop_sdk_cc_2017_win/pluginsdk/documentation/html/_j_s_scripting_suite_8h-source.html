00105 /*******************************************************************************
00106  * @ingroup JavaScriptSuite
00107 
00108     How to use the Scripting suite:
00109     
00110     To execute a JavaScript script inside a host application, you must acquire 
00111     and use a JavaScript engine instance.
00112 
00113     A typical use of the engine:
00114 
00115         JSEngineRef     engineRef = NULL;
00116         SPErr           err = noErr;
00117 
00118         JSScriptingSuite2 * sScriptingSuite = NULL;
00119         err = basicSuite->AcquireSuite(kJSScriptingSuite, 
00120                                    kJSScriptingSuiteVersion, 
00121                                    (const void **)&sScriptingSuite);
00122         if ( err == kSPNoError ) 
00123         {
00124             err = sScriptingSuite->CreateEngine( &engineRef );
00125             if ( err == kSPNoError )
00126             {
00127                 const ASUnicode        scriptText[] = {'a', 'l', 'e', 'r', 't', '(', '\'', 'h', 'i', '\'', ')', ';', 0 } ;  //    Script to create a new document
00128                 const ASUnicode        * result = NULL;
00129             
00130                 err = sScriptingSuite->ExecuteScript( engineRef, scriptText, kJSExecutionModeNoDebug, &result );
00131                 if ( err == kJSScriptExecutionError )
00132                 {
00133                     //  If error during execution, display a dialog containing returned error information string.
00134                     // ErrorAlert( result );
00135                 }
00136             }
00137         }
00138         
00139         ...
00140         
00141         err = sScriptingSuite->DeleteEngine( engineRef );
00142         engineRef = NULL;
00143 
00144         basicSuite->ReleaseSuite(kJSScriptingSuite, 
00145                                    kJSScriptingSuiteVersion);
00146 
00147 
00148     Notes:
00149     
00150         Do not deallocate the result value.  The engine instance owns it and will dispose of it.
00151 
00152         The engine instance doesn't need to be deleted after calling ExecuteJavaScript.  If you execute 
00153         multiple JavaScript scripts inside the same engine instance they will share the same context 
00154         (including global variables).  If you want scripts to execute independently use a different 
00155         engine instance for each script.
00156 */
